Usage Examples

Python - Get Main Dashboard Code

import requests import base64 # Method 1: Direct raw access url = "https://raw.githubusercontent.com/nourimabrouk/Een/main/metastation_pro.py" response = requests.get(url) code = response.text # Method 2: GitHub API api_url = "https://api.github.com/repos/nourimabrouk/Een/contents/metastation_pro.py" headers = {"Accept": "application/vnd.github.v3+json"} response = requests.get(api_url, headers=headers) content_b64 = response.json()["content"] code = base64.b64decode(content_b64).decode('utf-8')

// Direct raw access fetch('https://raw.githubusercontent.com/nourimabrouk/Een/main/metastation_pro.py') .then(response => response.text()) .then(code => console.log(code)); // GitHub API fetch('https://api.github.com/repos/nourimabrouk/Een/contents/metastation_pro.py') .then(response => response.json()) .then(data => { const code = atob(data.content); console.log(code); });
